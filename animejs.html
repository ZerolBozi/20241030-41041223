anime.js 動畫效果

anime.js 官網

範例程式碼

引入CDN 
<!-- 引入必要的腳本 -->
<script src="https://cdnjs.cloudflare.com/ajax/libs/animejs/3.2.1/anime.min.js"></script>
JS程式碼 
// 創建一個基本動畫
anime({
    targets: '.box1',
    translateX: 250,
    rotate: '1turn',
    duration: 800,
    easing: 'easeInOutQuad'
});

// 創建時間軸動畫
const timeline = anime.timeline({
    easing: 'easeOutExpo',
    duration: 750
});

// 添加動畫序列
timeline
    .add({
        targets: '.box2',
        scale: [0, 1],
        opacity: [0, 1]
    })
    .add({
        targets: '.box2',
        translateX: 250,
        rotate: 360
    })
    .add({
        targets: '.box2',
        translateY: [-50, 50],
        direction: 'alternate',
        loop: true
    });

// 創建無限循環動畫
anime({
    targets: '.box3',
    translateX: 250,
    translateY: [-50, 50],
    backgroundColor: ['#0d6efd', '#dc3545'],
    borderRadius: ['0%', '50%'],
    easing: 'easeInOutQuad',
    direction: 'alternate',
    loop: true,
    duration: 1500
});
