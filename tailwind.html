Tailwind CSS 樣式效果

Tailwind CSS 官網

範例程式碼

引入CDN 
<!-- 引入必要的腳本 -->
<script src="https://cdn.tailwindcss.com"></script>
樣式語法
<!-- 基礎樣式 -->
<div class="bg-blue-500 w-16 h-16 rounded shadow-lg hover:bg-blue-600 transition-all duration-300">
</div>

<!-- 漸變效果 -->
<div class="bg-gradient-to-r from-cyan-500 to-blue-500 w-16 h-16 rounded-lg hover:scale-110 transition-transform duration-300">
</div>

<!-- 動畫效果 -->
<div class="bg-blue-500 w-16 h-16 rounded-full animate-bounce">
</div>

<!-- 複合效果 -->
<div class="bg-blue-500 w-16 h-16 rounded-lg hover:bg-blue-600 hover:rotate-45 hover:scale-110 transition-all duration-300 ease-in-out shadow-lg hover:shadow-xl">
</div>

<!-- 群組效果 -->
<div class="group bg-white p-4 hover:bg-gray-50 transition-all duration-300">
    <div class="bg-blue-500 w-16 h-16 rounded group-hover:bg-red-500 transition-colors duration-300">
    </div>
</div>
